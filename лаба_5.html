Задание 6

function isPrime(num) {
    if (num < 2) return false;
    for (let i = 2; i <= Math.sqrt(num); i++) {
        if (num % i === 0) return false;
    }
    return true;
}
let n = 10;
let primes = [];
for (let i = 2; i <= n; i++) {
    if (isPrime(i)) primes.push(i);
}
document.write("Простые числа: " + primes.join(", "));
